\section{Tweak Types}

\begin{itemize}
    \item[] \texttt{0x00}: Stateful WOTS+C chains (SF\_WOTS\_CHAIN)
    \item[] \texttt{0x01}: Stateful tree nodes (SF\_TREE)
    \item[] \texttt{0x02}: Stateless WOTS+C chains (SL\_WOTS\_CHAIN)
    \item[] \texttt{0x03}: Stateless tree nodes (SL\_TREE)
    \item[] \texttt{0x04}: FORS/PORS (FORS)
    \item[] \texttt{0x05}: Message hash (H\_MSG)
    \item[] \texttt{0x06}: Pseudorandom function (PRF)
    \item[] \texttt{0x07}: Pseudorandom function for the message (PRF\_MSG)
    \item[] \texttt{0x08}: Root public key (ROOT)
\end{itemize}
