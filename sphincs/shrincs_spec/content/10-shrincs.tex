\section{SHRINCS}
This section specifies the complete SHRINCS signature scheme by composing the building blocks defined in previous sections.

\subsection{Key Generation}
Generates a fresh SHRINCS key pair and initializes the signing state.

\begin{verbatim}
    Algorithm: SHRINCS.KeyGen()
    Output:
        SK: secret key
        PK: public key
        state: initial signing state

    1. seed ←$ {0,1}^(3n)
    2. SK.seed ← seed[0:n]
    3. SK.prf ← seed[n:2n]
    4. PK.seed ← seed[2n:3n]
    5. ADRS ← new_ADRS()
    6. PK.sf ← uxmss_root(SK.seed, PK.seed, ADRS)
    7. setLayerAddress(ADRS, d - 1)
    8. setTreeAddress(ADRS, 0)
    9. PK.sl ← xmss_root(SK.seed, PK.seed, ADRS, h_prime)
    10. setTypeAndClear(ADRS, ROOT) 
    11. PK.root ← Tw(PK.seed, ADRS, PK.sf || PK.sl)
    12. SK ← (SK.seed, SK.prf, PK.seed, PK.sf, PK.sl, PK.root)
    13. PK ← (PK.seed, PK.root)
    14. state ← { q: 0, valid: true }
    15. return (SK, PK, state)
\end{verbatim}

\subsection{Recovery from Seed}
Restores keys from a backed-up seed. Since the signing state is unknown, stateful signing is disabled and only stateless signing is available.
\begin{verbatim}
    Algorithm: SHRINCS.Restore(seed)
    Input:
        seed: master seed (3n bytes)
    Output:
        SK: secret key
        PK: public key
        state: invalid state (stateless only)

    1. SK.seed ← seed[0:n]
    2. SK.prf ← seed[n:2n]
    3. PK.seed ← seed[2n:3n]
    4. ADRS ← new_ADRS()
    5. PK.sf ← uxmss_root(SK.seed, PK.seed, ADRS)
    6. setLayerAddress(ADRS, d - 1)
    7. setTreeAddress(ADRS, 0)
    8. PK.sl ← xmss_root(SK.seed, PK.seed, ADRS, h')
    9. setTypeAndClear(ADRS, ROOT)
    10. PK.root ← Tw(PK.seed, ADRS, PK.sf || PK.sl)
    11. SK ← (SK.seed, SK.prf, PK.seed, PK.sf, PK.sl, PK.root)
    12. PK ← (PK.seed, PK.root)
    13. state ← { q: $\bot$, valid: false }
    14. return (SK, PK, state)
\end{verbatim}

\subsection{Statefull signing}
Signs a message using the efficient stateful (UXMSS) path. Requires valid state.
\begin{verbatim}
    Algorithm: SHRINCS.SignStateful(m, SK, state)
    Input:
        m: message to sign
        SK: secret key (SK.seed, SK.prf, PK.seed, PK.sf, PK.sl, PK.root)
        state: current signing state { q, valid }
    Output:
        sig: stateful signature, or $\bot$ if state exhausted
        state': updated state

    1. if not state.valid:
        return ($\bot$, state)
    2. q ← state.q + 1
    3. if q > hsf + 1:
        return ($\bot$, state)
    4. ADRS ← new_ADRS()
    5. uxmss_sig ← uxmss_sign(m, SK.seed, SK.prf, PK.seed, PK.root, ADRS, q)
    6. sig ← uxmss_sig || PK.sl
    7. state' ← { q: q, valid: true }
    8. return (sig, state')
\end{verbatim}

\subsection{Stateless signing}
Signs a message using the stateless (SPHINCS+C) path. Used when state is lost or exhausted.
\begin{verbatim}
    Algorithm: SHRINCS.SignStateless(m, SK)
    Input:
        m: message to sign
        SK: secret key (SK.seed, SK.prf, PK.seed, PK.sf, PK.sl, PK.root)
    Output:
        sig: stateless signature

    1. ADRS ← new_ADRS()
    2. opt_rand ← random(n) or PK.seed
    3. R ← PRF_msg(SK.prf, PK.seed, opt_rand, m)
    4. setTypeAndClear(ADRS, H_MSG)
    5. digest ← H_msg(ADRS, R, 0, PK.seed, PK.root, m)
    6. (tree_idx, leaf_idx) ← parse_idx(digest, hsl, d)
    7. setLayerAddress(ADRS, 0)
    8. setTreeAddress(ADRS, tree_idx[0])
    9. fors_sig ← fors_sign(m, SK.seed, SK.prf, PK.seed, PK.root, ADRS)
    10. fors_pk ← fors_pkFromSig(fors_sig, m, PK.seed, PK.root, ADRS)
    11. ht_sig ← []
    12. msg ← fors_pk
    13. for layer from 0 to d - 1:
        a. setLayerAddress(ADRS, layer)
        b. setTreeAddress(ADRS, tree_idx[layer])
        c. xmss_sig ← xmss_sign(msg, SK.seed, SK.prf, PK.seed, PK.root, ADRS, h', leaf_idx[layer])
        d. ht_sig ← ht_sig || xmss_sig
        e. if layer < d - 1:
               msg ← xmss_root(SK.seed, PK.seed, ADRS, h')
    14. sig ← PK.sf || fors_sig || ht_sig
    15. return sig
\end{verbatim}

\subsection{Index Parsing}
Extracts tree and leaf indices for each hypertree layer from the message digest.

\begin{verbatim}
    Algorithm: parse_idx(digest, hsl, d)
    Input:
        digest: message digest
        hsl: total hypertree height
        d: number of layers
    Output:
        tree_idx: array of d tree indices
        leaf_idx: array of d leaf indices

    1. h_prime ← hsl / d 
    2. tree_idx ← []
    3. leaf_idx ← []
    4. idx ← bytes_to_int(digest[0 : (hsl + 7) / 8])
    5. for layer from 0 to d - 1:
       a. leaf_idx[layer] ← idx AND (2^h_prime - 1)
       b. idx ← idx >> h_prime
       c. tree_idx[layer] ← idx AND (2^(hsl - (layer + 1) * h_prime) - 1)
    6. return (tree_idx, leaf_idx)
\end{verbatim}

\subsection{Statefull verification}
\begin{verbatim}
    Algorithm: SHRINCS.VerifyStateful(m, sig, PK)
    Input:
        m: message
        sig: stateful signature
        PK: public key (PK.seed, PK.root)
    Output:
        valid: boolean
    
    1. ADRS ← new_ADRS()
    2. sig_len ← len(sig)
    3. PK.sl ← sig[sig_len - n : sig_len]
    4. uxmss_sig ← sig[0 : sig_len - n]
    5. wots_sig_size ← R_SIZE + c + l * n
    6. auth_len ← len(uxmss_sig) - wots_sig_size
    7. q ← auth_len / n
    8. if q < 1 or q > hsf + 1:
       return false
    9. wots_sig ← uxmss_sig[0 : wots_sig_size]
    10. auth ← uxmss_sig[wots_sig_size : wots_sig_size + q * n]
    11. PK.sf ← uxmss_pkFromSig(wots_sig, auth, m, PK.seed, PK.root, ADRS, q)
    12. if PK.sf == $\bot$:
        return false
    13. setTypeAndClear(ADRS, ROOT)
    14. expected_root ← Tw(PK.seed, ADRS, PK.sf || PK.sl)
    15. return (expected_root == PK.root)
\end{verbatim}

\subsection{Stateless verification}
\begin{verbatim}
    Algorithm: SHRINCS.VerifyStateless(m, sig, PK)
    Input:
        m: message
        sig: stateless signature
        PK: public key (PK.seed, PK.root)
    Output:
        valid: boolean

    1. ADRS ← new_ADRS()
    2. PK.sf ← sig[0 : n]
    3. offset ← n
    4. fors_sig_size ← R_SIZE + c + (k - 1) * (n + a * n) + n
    5. fors_sig ← sig[offset : offset + fors_sig_size]
    6. offset ← offset + fors_sig_size
    7. R ← fors_sig[0 : R_SIZE]
    8. ctr ← fors_sig[R_SIZE : R_SIZE + c]
    9. setTypeAndClear(ADRS, H_MSG)
    10. digest ← H_msg(ADRS, R, ctr, PK.seed, PK.root, m)
    11. (tree_idx, leaf_idx) ← parse_idx(digest, hsl, d)
    12. setLayerAddress(ADRS, 0)
    13. setTreeAddress(ADRS, tree_idx[0])
    14. fors_pk ← fors_pkFromSig(fors_sig, m, PK.seed, PK.root, ADRS)
    15. if fors_pk == $\bot$:
            return false
    16. msg ← fors_pk
    17. xmss_sig_size ← R_SIZE + c + l * n + h' * n
    18. for layer from 0 to d - 1:
        a. xmss_sig ← sig[offset : offset + xmss_sig_size]
        b. offset ← offset + xmss_sig_size
        c. wots_sig ← xmss_sig[0 : R_SIZE + c + l * n]
        d. auth ← xmss_sig[R_SIZE + c + l * n : ]
        e. setLayerAddress(ADRS, layer)
        f. setTreeAddress(ADRS, tree_idx[layer])
        g. msg ← xmss_pkFromSig(wots_sig, auth, msg, PK.seed, PK.root, ADRS, h', leaf_idx[layer])
        h. if msg == $\bot$:
               return false
    19. PK.sl ← msg
    20. setTypeAndClear(ADRS, ROOT)
    21. expected_root ← Tw(PK.seed, ADRS, PK.sf || PK.sl)
    22. return (expected_root == PK.root)
\end{verbatim}

\subsection{Unified Verification}
\begin{verbatim}
    Algorithm: SHRINCS.Verify(m, sig, PK)
    Input:
        m: message
        sig: signature (stateful or stateless)
        PK: public key (PK.seed, PK.root)
    Output:
        valid: boolean

1. if len(sig) <= max_sf_size:
       return SHRINCS.VerifyStateful(m, sig, PK)
   else:
       return SHRINCS.VerifyStateless(m, sig, PK)
\end{verbatim}
